        Random(ca,1,5)

  robot Coordinar
  variables
    Id:numero
    BolsaP:numero
    BolsaF:numero
    bolsa:numero
  comenzar
    EnviarMensaje(1,F1)
    EnviarMensaje(2,F2)
    EnviarMensaje(3,P1)
    EnviarMensaje(4,P2)
    repetir 2
      RecibirMensaje(Id,*)
    BolsaP:=0
    BolsaF:=0
    mientras(BolsaP<2)&(BolsaF<2)
      EnviarMensaje(0,F2)
      EnviarMensaje(0,P2)
      repetir 2
        RecibirMensaje(Id,*)
        si(Id=2)
          RecibirMensaje(bolsa,F2)
          BolsaF:=BolsaF+bolsa
        sino
          RecibirMensaje(bolsa,P2)
          BolsaP:=BolsaP+bolsa
    EnviarMensaje(1,F2)
    EnviarMensaje(1,P2)
    si(BolsaF<BolsaP)
      Informar(1)
    sino
      Informar(2)
  fin
