areas
  areapc: AreaPC(11,11,11,11)
  areap1: AreaP(4,1,4,100)
  areap2: AreaP(6,1,6,100)
  areap3: AreaP(8,1,8,100)
  areap4: AreaP(10,1,10,100)
  areapFisc: AreaP(1,1,1,1)
robots
  robot Racer
  variables
    Id:numero
    fini:numero
    ActualCa:numero
    ActualAv:numero
  comenzar
    RecibirMensaje(Id,Fiscal)
    ActualCa:=PosCa
    ActualAv:=PosAv
    fini:=1
    BloquearEsquina(11,11)
    Pos(11,11)
    si(HayPapelEnLaEsquina)
      tomarPapel
      Pos(ActualAv,ActualCa)
      LiberarEsquina(11,11)
      depositarPapel
      mover
    sino
      Pos(ActualAv,ActualCa)
      LiberarEsquina(11,11)
      fini:=0
    mientras(fini=1)&(PosCa<100)
      ActualCa:=PosCa
      ActualAv:=PosAv
      BloquearEsquina(11,11)
      Pos(11,11)
      si(HayPapelEnLaEsquina)
        tomarPapel
        Pos(ActualAv,ActualCa)
        LiberarEsquina(11,11)
        depositarPapel
        mover
      sino
        Pos(ActualAv,ActualCa)
        LiberarEsquina(11,11)
        fini:=0
    EnviarMensaje(Id,Fiscal)
    EnviarMensaje(ActualCa,Fiscal)
  fin
  robot Servidor
  variables
    Id:numero
    ganador:numero
    calles1:numero
    calles2:numero
    calles3:numero
    calles4:numero
  comenzar
    EnviarMensaje(1,R1)
    EnviarMensaje(2,R2)
    EnviarMensaje(3,R3)
    EnviarMensaje(4,R4)
    repetir 4
      RecibirMensaje(Id,*)
      si(Id=1)
        RecibirMensaje(calles1,R1)
      sino
        si(Id=2)
          RecibirMensaje(calles2,R2)
        sino
          si(Id=3)
            RecibirMensaje(calles3,R3)
          sino
            RecibirMensaje(calles4,R4)
    si(calles1>calles2)&(calles1>calles3)&(calles1>calles4)
      ganador:=1
    sino
      si(calles2>calles1)&(calles2>calles3)&(calles2>calles4)
        ganador:=2
      sino
        si(calles3>calles1)&(calles3>calles2)&(calles3>calles4)
          ganador:=3
        sino
          si(calles4>calles1)&(calles4>calles2)&(calles4>calles3)
            ganador:=4
    Informar(ganador)   
  fin
variables
  R1:Racer
  R2:Racer
  R3:Racer
  R4:Racer
  Fiscal:Servidor
comenzar
  AsignarArea(R1,areap1)
  AsignarArea(R1,areapc)
  AsignarArea(R2,areap2)
  AsignarArea(R2,areapc)
  AsignarArea(R3,areap3)
  AsignarArea(R3,areapc)
  AsignarArea(R4,areap4)
  AsignarArea(R4,areapc)
  AsignarArea(Fiscal,areapFisc)
  Iniciar(Fiscal,1,1)
  Iniciar(R1,4,1)
  Iniciar(R2,6,1)
  Iniciar(R3,8,1)
  Iniciar(R4,10,1)
fin
